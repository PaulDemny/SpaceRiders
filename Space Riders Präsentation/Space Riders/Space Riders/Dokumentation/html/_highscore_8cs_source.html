138  public string HighscoreLesen(string commando)
139  {
140  try
141  {
142  //Verbindung zur Datenbank welche, die Highscore beinhaltet, wird aufgebaut
143  verbindung.ConnectionString = "Data Source = " + datenquelle;
144  verbindung.Open();
145  SQLiteCommand command = new SQLiteCommand(verbindung);
146  //Kommandotext, der den Namen, den Highscore und die Schwierigkeit von Spielern aus der Datenbank ausliest, abhängig vom string commando (Einschränkung)
147  command.CommandText = "Select Name, Score, Schwierigkeit from Highscore " + commando + " order by Score desc";
148  command.ExecuteNonQuery();
149  //Die Datenbak wird ausgelesen
150  SQLiteDataReader reader = command.ExecuteReader();
151  //Dieser string wird mit den Datensätzen der Datenbak gefüllt
152  string erg = "";
153  //Diese Variable beinhaltet die Platzierung
154  int platz = 1;
155  //Der Reader liest Zeile für Zeile
156  if (reader.HasRows)
157  while (reader.Read())
158  {
159  //Platzierung
160  erg += platz.ToString() + ". Platz: ";
161  //Name des Spielers
162  erg += reader.GetString(reader.GetOrdinal("Name"));
163  //Punktestand des Spielers
164  erg += ", Score: ";
165  erg += reader.GetInt32(reader.GetOrdinal("Score"));
166  //Falls die gesamte Datenbank ausgelesen wird, wird die Schwierigkeit angezeigt, sonst nicht, da die Auswahl der Schwierigkeit vom User durch die Combobox fesetgelegt
167  if (commando == "")
168  {
169  //Schwierigkeitsstufe
170  erg += ", Schwierigkeit: ";
171  erg += reader.GetString(reader.GetOrdinal("Schwierigkeit"));
172  }
173  //der string wird der Anzeige (Listbox) hinzugefügt
174  this.lbHighscore.Items.Add(erg);
175  //der string wird geleert, um den nächsten DAtensatz einzulesen
176  erg = "";
177  //Die Platzierung wird um eins erhöht, d.h auf den näcchst schlechteren Spieler gesetzt
178  platz++;
179  }
180  //Die Verbindung zur Datenbank wird geschlossen
181  verbindung.Close();
182  return erg;
183  }
184  catch (Exception ex)
185  {
186  string nichts = "";
187  MessageBox.Show("Highscore konnte nicht angezeigt werden");
188  return nichts;
189  }
